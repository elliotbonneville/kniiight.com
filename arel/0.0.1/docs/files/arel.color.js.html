 (function() {

 		"gray": "808080",
 		"black": "000000",
 		"red": "FF0000",
 		"maroon": "800000",
 		"yellow": "FFFF00",
 		"olive": "808000",
 		"lime": "00FF00",

 		"aqua": "00FFFF",
 		"teal": "008080",
 		"blue": "0000FF",
 		"navy": "000080",
 		"fuschia": "FF00FF",

 	 * @param g Green component of an RGBA color.

 	/**
 	 * Returns a color.
 	 *
 	 * @method color
 	 * @for arel
 	 * 
 	 * @param color One of the 16 standard colors supported by all browsers.
 	 * 
 	 * @return {RGBAColor}
 	 * 
 	 */
 	 
 	 /**
 	 * Returns a color.
 	 *
 	 * @method color
 	 * @for arel
 	 * 
 	 * @param color Hexadecimal representation of a color.
 	 * 
 	 * @return {RGBAColor}
 	 * 
 	 */
 	function color(r, g, b, a) {
 		if(arguments.length == 1) {
 			// if the first argument is a hexadecimal color
 			if(/(^#[0-9A-F]{6}$)|(^#[0-9A-F]{3}$)/i.test(r)) {
 				return colorFromHex(r);
 			} else if(typeof colors[r] !== "undefined") {
 				return colorFromHex(colors[r]);
 			}
 		} else {
 			
 		}
 		
 		throw new Error("Invalid color supplied!");
 	}
 	
 	// http://stackoverflow.com/a/5624139/339852
 	function colorFromHex(hex) {
 		// Expand shorthand form (e.g. "03F") to full form (e.g. "0033FF")
	    var shorthandRegex = /^#?([a-f\d])([a-f\d])([a-f\d])$/i;
	    hex = hex.replace(shorthandRegex, function(m, r, g, b) {
	        return r + r + g + g + b + b;
	    });

	    var result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
	    
	    if(!result) {
	    	throw new Error("Invalid color supplied!");
	    }
	    
	    return colorFromRGBA(parseInt(result[1], 16), parseInt(result[2], 16), parseInt(result[3], 16));
 	}
 	
 	function colorFromRGBA(r, g, b, a) {
 		return new RGBAColor(r, g, b, a);
 	}
 	
 	/**
 	 * Stores an RGBA color
 	 * 
 	 * @class RGBAColor
 	 * @constructor
 	 * @param r
 	 * @param g
 	 * @param b
 	 * @param [a]
 	 */
 	function RGBAColor(r, g, b, a) {
 		/**
 		 * The red component of this color (range 0 to 255).
 		 * 
 		 * @property r
 		 * @type Number
 		 */
 		this.r = r || 0;
 		
 		/**
 		 * The green component of this color (range 0 to 255).
 		 * 
 		 * @property g
 		 * @type Number
 		 */
 		this.g = g || 0;
 		
 		/**
 		 * The blue component of this color (range 0 to 255).
 		 * 
 		 * @property b
 		 * @type Number
 		 */
 		this.b = b || 0;
 		
 		/**
 		 * The alpha component of this color (range 0 to 1).
 		 * 
 		 * @property a
 		 * @type Number
 		 */
 		this.a = a || 1;
 	}
 	
 	/**
 	 * Returns the string value of this color in the format "rgba(r, g, b, a)".
 	 * 
 	 * @method toString
 	 * @return String
 	 */
 	RGBAColor.prototype.toString = function() {
 		return "rgba(" + this.r + ", " + this.g + ", " + this.b + ", " + this.a + ")";
 	}
 	
 	arel.color = color;
})();
