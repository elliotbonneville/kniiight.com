/**
 * The namespace for the entire library
 *
 * @module arel
 */

(function() {
	/**
	 * Global namespace
	 *
	 * @class arel
	 * @static
	 */
	var arel = {};
	
	/**
	 * Initiates Arel and all its modules. Should not be called until the document has loaded.
	 *
	 * @method init
	 * @param {Object} parameters
	 *   @param parameters.canvas The canvas element to render to.
	 *   @param parameters.width The width in cells of the map display.
	 *   @param parameters.height The height in cells of the map display.
	 *   @param [parameters.font] The font used in rendering (web-font enabled).
	 *   @param [parameters.fontSize] The font size used to render.
	 *   @param [parameters.padding] The padding between a rendered character and the borders of the cell it occupies.
	 *   @param [parameters.batchingEnabled] Whether or not to enable render batching. (Set to true by default.) Render batching increases render speed by 500% to 600%.
	 * @return {Object} Arel
	 */
	arel.init = function(parameters) {
		if(typeof parameters === "undefined") {
			throw new Error("Please specify Arel session parameters.");
		}
		
		// check if a canvas element has been passed; if not, fail
		if(parameters.canvas === undefined) {
			throw new Error("Arel needs a <canvas> element to render to.");
		}
		
		// initialize modules
		arel.display.__init.call(arel.display, this, parameters);
		
		return arel;
	}
	
	window.arel = arel;
})();
